如何实现圆环的中间显示一个带颜色的圆点图例

实现的思路是通过 label 的 formatter 把 label 替换成“●”，再在 data 中添加颜色

const option = {
    tooltip: {
       show: false,
    },
    legend: {
        show: false,
    },
    series: [
        {
            type: 'pie',
            center: ['50%', '50%'],
            radius: ['60%', '70%'],
            avoidLabelOverlap: false,
            startAngle: 90,
            label: {
                show: true,
                position: 'outside',
                fontSize: 50,
                formatter: '●',
            },
            emphasis: {
                label: {
                    show: false,
                },
            },
            labelLine: {
                show: false,
                length: 5,
                length2: 0,
            },
            data: [
                {
                    name: '90后',
                    value: 50000,
                    label: {
                        color: '#5FFFAA',
                    },
                },
                {
                    name: '80后',
                    value: 20000,
                    label: {
                        color: '#FFBC3E',
                    },
                },
                {
                    name: '70后以上',
                    value: 30000,
                    label: {
                        color: '#139BFF',
                    },
                },
            ],
            itemStyle: {
                borderWidth: 0,
                borderColor: '#1f3a54',
            },
            color: [
                {
                    type: 'linear',
                    x: 0,
                    y: 0,
                    x2: 1,
                    y2: 0,
                    colorStops: [
                        {
                            offset: 0,
                            color: '#5FFFAA',
                        },
                        {
                            offset: 1,
                            color: '#AAFFD2',
                        },
                    ],
                    global: false,
                },
                {
                    type: 'linear',
                    x: 0,
                    y: 0,
                    x2: 1,
                    y2: 0,
                    colorStops: [
                        {
                            offset: 0,
                            color: '#FFDB97',
                        },
                        {
                            offset: 1,
                            color: '#FFBC3E',
                        },
                    ],
                    global: false,
                },
                {
                    type: 'linear',
                    x: 0,
                    y: 0,
                    x2: 1,
                    y2: 0,
                    colorStops: [
                        {
                            offset: 0,
                            color: '#139BFF',
                        },
                        {
                            offset: 1,
                            color: '#0AFFF5',
                        },
                    ],
                    global: false,
                },
            ],
        },
    ],
}
